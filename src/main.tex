\newcommand{\abspos}[3]{%
  \begin{tikzpicture}[remember picture, overlay]

% ---- 封面页（标题页、致谢页共用）----
% 用法: \coverslide{中心内容}
\newcommand{\coverslide}[1]{%
  \setbeamertemplate{background}{}%
  \begin{frame}[plain]
    \begin{tikzpicture}[remember picture, overlay]
      \draw[accentTwo, line width=1.5pt]
        ([xshift=\borderInset, yshift=-\borderTopY]current page.north west)
        rectangle
        ([xshift=-\borderInset, yshift=\borderBotY]current page.south east);
    \end{tikzpicture}
    \abspos{\borderInset}{\borderTopY}{%
      \includegraphics[width=\logoW, height=\logoH]{logo.png}%
    }
    #1
    \begin{tikzpicture}[remember picture, overlay]
      \fill[accentOne]
        ([xshift=6.667in, yshift=-\bottomBarY]current page.north west)
        .. controls ([xshift=9.133in, yshift=-\bottomBarY]current page.north west)
                    and ([xshift=11.367in, yshift={-(\bottomBarY+0.152in)}]current page.north west)
        .. ([xshift=12.981in, yshift={-(\bottomBarY+0.397in)}]current page.north west)
        -- ([xshift=\slideW, yshift={-(\bottomBarY+0.455in)}]current page.north west)
        -- ([xshift=\slideW, yshift={-(\bottomBarY+\bottomBarH)}]current page.north west)
        -- ([xshift=0in, yshift={-(\bottomBarY+\bottomBarH)}]current page.north west)
        -- ([xshift=0in, yshift={-(\bottomBarY+0.455in)}]current page.north west)
        -- ([xshift=0.350in, yshift={-(\bottomBarY+0.397in)}]current page.north west)
        .. controls ([xshift=1.967in, yshift={-(\bottomBarY+0.152in)}]current page.north west)
                    and ([xshift=4.200in, yshift=-\bottomBarY]current page.north west)
        .. cycle;
    \end{tikzpicture}
  \end{frame}%
}

\newcommand{\navbar}[1]{%

    \abspos{0in}{0in}{\parbox[c][\navbarH]{\navSectionOneW}{\centering{\fontsize{24}{29}\selectfont\bfseries\color{white}\hspace{\navActivePad}功能演示}}}%
  \else
    \abspos{0in}{0in}{\parbox[c][\navbarH]{\navSectionOneW}{\centering{\fontsize{20}{24}\selectfont\color{white}\hspace{\navInactivePad}功能演示}}}%
  \fi
  %
  \abspos{\navDivX}{\navDivY}{\color{white}\rule{0.6pt}{\navDivH}}%
  %
  \ifnum#1=2
    \abspos{\navDivX}{0in}{\parbox[c][\navbarH]{\navSectionTwoW}{\centering{\fontsize{24}{29}\selectfont\bfseries\color{white}考核题目讲解}}}%

    \abspos{\navDivX}{0in}{\parbox[c][\navbarH]{\navSectionTwoW}{\centering{\fontsize{20}{24}\selectfont\color{white}考核题目讲解}}}%
